102 void ObjectTracking::trackFilteredObject(int &x, int &y, Mat threshold, Mat &cameraFeed)
103 {
104  Mat temp;
105  threshold.copyTo(temp);
106  //these two vectors needed for output of findContours
107  vector< vector<Point> > contours;
108  vector<Vec4i> hierarchy;
109  //find contours of filtered image using openCV findContours function
110  findContours(temp,contours,hierarchy,CV_RETR_CCOMP,CV_CHAIN_APPROX_SIMPLE );
111  //use moments method to find our filtered object
112  double refArea = 0;
113  bool objectFound = false;
114  Mat dst = Mat::zeros(temp.size(), CV_8UC3);
115  if (hierarchy.size() > 0) {
116  int numObjects = hierarchy.size();
117  //if number of objects greater than MAX_NUM_OBJECTS we have a noisy filter
118  if(numObjects<MAX_NUM_OBJECTS){
119  for (int index = 0; index >= 0; index = hierarchy[index][0]) {
120 
121  Moments moment = moments((cv::Mat)contours[index]);
122  double area = moment.m00;
123 
124  //if the area is less than 20 px by 20px then it is probably just noise
125  //if the area is the same as the 3/2 of the image size, probably just a bad filter
126  //we only want the object with the largest area so we safe a reference area each
127  //iteration and compare it to the area in the next iteration.
128  if(area>MIN_OBJECT_AREA && area<MAX_OBJECT_AREA){
129  x = moment.m10/area;
130  y = moment.m01/area;
131  objectFound = true;
132  refArea = area;
133  }else objectFound = false;
134 
135 
136  }
137  //let user know you found an object
138  if(objectFound ==true){
139  justObjFound = true;
140  putText(cameraFeed,"Tracking Object",Point(0,50),2,1,Scalar(0,255,0),2);
141  //draw object location on screen
142  drawObject(x,y,cameraFeed);
143  if(shouldCreate)
144  {
145  //these for built a directory and a file
146  wxDateTime DT = wxDateTime::Now();
147  wxString str4date = DT.FormatISODate();
148  wxString str4time = DT.FormatISOTime();
149  str4time.Replace(":","-");
150  wxString str4dir=_T("D:\\CamTrackerVideo\\")+str4date;
151  wxFileName::SetCwd(_T("D:\\")); //we should set the current working directory before build a sub-directory
152  if(!wxFileName::DirExists(_T("CamTrackerVideo")))
153  {
154  if(!wxFileName::Mkdir(_T("CamTrackerVideo")))
155  {
156  wxMessageBox(_T("Fail to build directory CamTrackerVideo!"),_T("Error"));
157  }
158  }
159  wxFileName::SetCwd(_T("D:\\CamTrackerVideo"));
160  if(!wxFileName::DirExists(str4date))
161  {
162  if(!wxFileName::Mkdir(str4date))
163  {
164  wxMessageBox(_T("Fail to build directory by date!"),_T("Error"));
165  }
166  }
167  wxString name2save=str4dir+_T("\\")+str4time+".avi";
168  std::string Cname2save = std::string(name2save.mbc_str()); //exchange wxString to std::string
169  writer.open(Cname2save,CV_FOURCC('M','J','P','G'),FPS,Size(FRAME_WIDTH,FRAME_HEIGHT));

172  wxMessageBox(_T("Fail to open file!"),_T("Error"));

177  writer << cameraFeed;
178  }
179  /*else{

181  shouldCreate = true;
182  }*/
183

185  }

187  if(writer.isOpened())

210 void ObjectTracking::tracking(Mat &frame)
211 {
212  //store image to matrix
213  icap.read(cameraFeed);
214  FRAME_HEIGHT = icap.get(CV_CAP_PROP_FRAME_HEIGHT);
215  FRAME_WIDTH = icap.get(CV_CAP_PROP_FRAME_WIDTH);
216  //convert frame from BGR to HSV colorspace
217  cvtColor(cameraFeed,HSV,COLOR_BGR2HSV);
218  //filter HSV image between values and store filtered image to
219  //threshold matrix
220  inRange(HSV,Scalar(H_MIN,S_MIN,V_MIN),Scalar(H_MAX,S_MAX,V_MAX),threshold);
221  //perform morphological operations on thresholded image to eliminate noise
222  //and emphasize the filtered object(s)
223  if(useMorphOps)
224  morphOps(threshold);
225  //pass in thresholded frame to our object tracking function
226  //this function will return the x and y coordinates of the
227  //filtered object
228  if(trackObjects)
229  {
230  trackFilteredObject(x,y,threshold,cameraFeed);
231  }
232  frame = cameraFeed;
233 }
